	\node at (3,-2) {$0$}; \node[honest] at (3,.7) (a2) {$3$};
	\node at (4,-2) {$1$}; \node[malicious] at (4,-.7) (b2) {$4$};
	\node at (5,-2) {$0$}; \node[honest] at (5,-.7) (b3) {$5$};
	\node at (6,-2) {$1$}; \node[malicious] at (6,.7) (a3) {$6$};
  	\node[honest] at (-1,0) (base) {$0$};

	    \draw[thick,->] (a1) -- (a2);
	    \draw[thick,->] (a2) -- (a3);

	    \draw[thick,->] (b1) -- (b2);
	    \draw[thick,->] (b2) -- (b3);
